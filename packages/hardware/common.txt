# Hardware support packages installed on every machine (setup/5-hardware.sh).
# GPU-specific packages are in separate files: gpu-nvidia.txt, gpu-amd.txt, gpu-intel.txt.
# Laptop-only packages are in laptop.txt.

## Wireless networking
iwd                 # Intel wireless daemon (modern Wi-Fi management)
wireless_tools      # Classic Wi-Fi tools: iwconfig, iwlist
wpa_supplicant      # WPA/WPA2 authentication backend

## Bluetooth
bluez               # Linux Bluetooth protocol stack
bluez-utils         # Bluetooth CLI tools (bluetoothctl)

## Audio - PipeWire stack (replaces PulseAudio and JACK)
pipewire            # Modern audio and video processing server
pipewire-alsa       # ALSA compatibility layer
pipewire-pulse      # PulseAudio compatibility layer
pipewire-jack       # JACK compatibility layer
wireplumber         # PipeWire session and policy manager

## Compressed swap
zram-generator      # Systemd-based zram compressed swap setup
