# NVIDIA proprietary driver stack.
# Uses DKMS so the module rebuilds automatically for any kernel update (linux-zen, etc.).
# After install: modprobe.d and mkinitcpio are configured by 5-hardware.sh.

dkms                # Dynamic Kernel Module System - auto-rebuilds modules on kernel upgrades
nvidia-dkms         # NVIDIA driver via DKMS (works with linux-zen and any other kernel)
nvidia-utils        # NVIDIA runtime libraries and utilities (nvidia-smi, etc.)
nvidia-settings     # NVIDIA graphical control panel
libva-nvidia-driver # VA-API support for NVIDIA (hardware-accelerated video decode/encode)
