# Packages installed via pacstrap during live ISO phase (setup/2-base.sh).
# Keep this list minimal: only what is needed to boot into the new system.
# Everything else goes in hardware/, desktop.txt, or apps.txt.

base                # Arch Linux base system (systemd, bash, coreutils, etc.)
linux-zen           # Zen kernel - optimized for desktop responsiveness
linux-zen-headers   # Kernel headers required by DKMS modules (NVIDIA, etc.)
linux-firmware      # Firmware blobs for common hardware (Wi-Fi cards, SSDs, etc.)
sudo                # Privilege escalation
networkmanager      # Network management daemon (enables internet after first boot)
vim                 # Terminal text editor (for emergency repairs)
